16/// Enum listing possible errors from duckdb.
17#[derive(Debug)]
18#[allow(clippy::enum_variant_names)]
19#[non_exhaustive]
20pub enum Error {
21    /// An error from an underlying DuckDB call.
22    DuckDBFailure(FFIError, Option<String>),
23
24    /// Error when the value of a particular column is requested, but it cannot
25    /// be converted to the requested Rust type.
26    // FromSqlConversionFailure(usize, Type, Box<dyn error::Error + Send + Sync + 'static>),
27
28    /// Error when DuckDB gives us an integral value outside the range of the
29    /// requested type (e.g., trying to get the value 1000 into a `u8`).
30    /// The associated `usize` is the column index,
31    /// and the associated `i64` is the value returned by SQLite.
32    IntegralValueOutOfRange(usize, i128),
33
34    /// Error converting a string to UTF-8.
35    Utf8Error(str::Utf8Error),
36
37    /// Error converting a string to a C-compatible string because it contained
38    /// an embedded nul.
39    NulError(::std::ffi::NulError),
40
41    /// Error when using SQL named parameters and passing a parameter name not
42    /// present in the SQL.
43    InvalidParameterName(String),
44
45    /// Error converting a file path to a string.
46    InvalidPath(PathBuf),
47
48    /// Error returned when an [`execute`](crate::Connection::execute) call
49    /// returns rows.
50    ExecuteReturnedResults,
51
52    /// Error when a query that was expected to return at least one row (e.g.,
53    /// for [`query_row`](crate::Connection::query_row)) did not return any.
54    QueryReturnedNoRows,
55
56    /// Error when the value of a particular column is requested, but the index
57    /// is out of range for the statement.
58    InvalidColumnIndex(usize),
59
60    /// Error when the value of a named column is requested, but no column
61    /// matches the name for the statement.
62    InvalidColumnName(String),
63
64    /// Error when the value of a particular column is requested, but the type
65    /// of the result in that column cannot be converted to the requested
66    /// Rust type.
67    // InvalidColumnType(usize, String, Type),
68
69    /// Error when a query that was expected to insert one row did not insert
70    /// any or insert many.
71    StatementChangedRows(usize),
72
73    /// Error available for the implementors of the
74    /// [`ToSql`](crate::types::ToSql) trait.
75    ToSqlConversionFailure(Box<dyn error::Error + Send + Sync + 'static>),
76
77    /// Error when the SQL is not a `SELECT`, is not read-only.
78    InvalidQuery,
79
80    /// Error when the SQL contains multiple statements.
81    MultipleStatement,
82    /// Error when the number of bound parameters does not match the number of
83    /// parameters in the query. The first `usize` is how many parameters were
84    /// given, the 2nd is how many were expected.
85    InvalidParameterCount(usize, usize),
86
87    /// Append Error
88    AppendError,
89
90    ConversionError(DuckDBConversionError),
91}

96impl PartialEq for Error {
97    fn eq(
98        &self,
99        other: &Error,
100    ) -> bool {
101        match (self, other) {
102            (Error::DuckDBFailure(e1, s1), Error::DuckDBFailure(e2, s2)) => e1 == e2 && s1 == s2,
103            (Error::IntegralValueOutOfRange(i1, n1), Error::IntegralValueOutOfRange(i2, n2)) => {
104                i1 == i2 && n1 == n2
105            },
106            (Error::Utf8Error(e1), Error::Utf8Error(e2)) => e1 == e2,
107            (Error::NulError(e1), Error::NulError(e2)) => e1 == e2,
108            (Error::InvalidParameterName(n1), Error::InvalidParameterName(n2)) => n1 == n2,
109            (Error::InvalidPath(p1), Error::InvalidPath(p2)) => p1 == p2,
110            (Error::ExecuteReturnedResults, Error::ExecuteReturnedResults) => true,
111            (Error::QueryReturnedNoRows, Error::QueryReturnedNoRows) => true,
112            (Error::InvalidColumnIndex(i1), Error::InvalidColumnIndex(i2)) => i1 == i2,
113            (Error::InvalidColumnName(n1), Error::InvalidColumnName(n2)) => n1 == n2,
114            // (Error::InvalidColumnType(i1, n1, t1), Error::InvalidColumnType(i2, n2, t2)) => {
115            //     i1 == i2 && t1 == t2 && n1 == n2
116            // }
117            (Error::StatementChangedRows(n1), Error::StatementChangedRows(n2)) => n1 == n2,
118            (Error::InvalidParameterCount(i1, n1), Error::InvalidParameterCount(i2, n2)) => {
119                i1 == i2 && n1 == n2
120            },
121            (..) => false,
122        }
123    }
124}

163impl fmt::Display for Error {
164    fn fmt(
165        &self,
166        f: &mut fmt::Formatter<'_>,
167    ) -> fmt::Result {
168        match *self {
169            Error::DuckDBFailure(ref err, None) => err.fmt(f),
170            Error::DuckDBFailure(_, Some(ref s)) => write!(f, "{s}"),
171            // Error::FromSqlConversionFailure(i, ref t, ref err) => {
172            //     if i != UNKNOWN_COLUMN {
173            //         write!(f, "Conversion error from type {t} at index: {i}, {err}")
174            //     } else {
175            //         err.fmt(f)
176            //     }
177            // }
178            Error::IntegralValueOutOfRange(col, val) => {
179                if col != UNKNOWN_COLUMN {
180                    write!(f, "Integer {val} out of range at index {col}")
181                } else {
182                    write!(f, "Integer {val} out of range")
183                }
184            },
185            Error::Utf8Error(ref err) => err.fmt(f),
186            Error::NulError(ref err) => err.fmt(f),
187            Error::InvalidParameterName(ref name) => write!(f, "Invalid parameter name: {name}"),
188            Error::InvalidPath(ref p) => write!(f, "Invalid path: {}", p.to_string_lossy()),
189            Error::ExecuteReturnedResults => {
190                write!(f, "Execute returned results - did you mean to call query?")
191            },
192            Error::QueryReturnedNoRows => write!(f, "Query returned no rows"),
193            Error::InvalidColumnIndex(i) => write!(f, "Invalid column index: {i}"),
194            Error::InvalidColumnName(ref name) => write!(f, "Invalid column name: {name}"),
195            // Error::InvalidColumnType(i, ref name, ref t) => {
196            //     write!(f, "Invalid column type {t} at index: {i}, name: {name}")
197            // }
198            // Error::ArrowTypeToDuckdbType(ref name, ref t) => {
199            //     write!(f, "Invalid column type {t} , name: {name}")
200            // }
201            Error::InvalidParameterCount(i1, n1) => {
202                write!(f, "Wrong number of parameters passed to query. Got {i1}, needed {n1}")
203            },
204            Error::StatementChangedRows(i) => write!(f, "Query changed {i} rows"),
205            Error::ToSqlConversionFailure(ref err) => err.fmt(f),
206            Error::InvalidQuery => write!(f, "Query is not read-only"),
207            Error::MultipleStatement => write!(f, "Multiple statements provided"),
208            Error::AppendError => write!(f, "Append error"),
209            Error::ConversionError(ref err) => match err {
210                DuckDBConversionError::TypeMismatch { expected, found } => {
211                    write!(f, "Type mismatch: expected {expected}, found {found}")
212                },
213                DuckDBConversionError::ConversionError(ref msg) => {
214                    write!(f, "Conversion error: {msg}")
215                },
216                DuckDBConversionError::NullValue => write!(f, "Null value encountered"),
217                DuckDBConversionError::PrecisionLoss(ref msg) => write!(f, "Precision loss: {msg}"),
218            },
219        }
220    }
221}

223impl error::Error for Error {
224    fn source(&self) -> Option<&(dyn error::Error + 'static)> {
225        match *self {
226            Error::DuckDBFailure(ref err, _) => Some(err),
227            Error::Utf8Error(ref err) => Some(err),
228            Error::NulError(ref err) => Some(err),
229
230            Error::IntegralValueOutOfRange(..)
231            | Error::InvalidParameterName(_)
232            | Error::ExecuteReturnedResults
233            | Error::QueryReturnedNoRows
234            | Error::InvalidColumnIndex(_)
235            | Error::InvalidColumnName(_)
236            // | Error::InvalidColumnType(..)
237            | Error::InvalidPath(_)
238            | Error::InvalidParameterCount(..)
239            | Error::StatementChangedRows(_)
240            | Error::InvalidQuery
241            | Error::AppendError
242            // | Error::ArrowTypeToDuckdbType(..)
243            | Error::MultipleStatement | Error::ConversionError(_) => None,
244            // Error::FromSqlConversionFailure(_, _, ref err)
245            Error::ToSqlConversionFailure(ref err) => Some(&**err),
246        }
247    }
