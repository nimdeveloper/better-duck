16/// Represents a prepared DuckDB statement, holding the connection, the prepared statement pointer, and execution result.
17pub struct Statement {
18    /// Shared reference to the underlying DuckDB connection.
19    con: Arc<RawConnection>,
20    /// Pointer to the prepared DuckDB statement (FFI resource).
21    stmt: duckdb_prepared_statement,
22    /// Result of the last execution, if any.
23    result: Option<RawResult>,
24    /// Index for parameter binding.
25    bind_idx: u64,
26}
27
28impl Statement {
29    /// Create a new prepared statement from a SQL string.
30    ///
31    /// # Arguments
32    ///
33    /// * `con` - Reference to a raw DuckDB connection.
34    /// * `sql` - SQL query string to prepare.
35    ///
36    /// # Returns
37    ///
38    /// * `Result<Self>` - The prepared statement or an error.
39    ///
40    /// # Example
41    ///
42    /// ```ignore
43    /// let con = get_test_connection();
44    /// let stmt = Statement::new(&con, "SELECT 1");
45    /// assert!(stmt.is_ok());
46    /// ```
47    pub(super) fn new(
48        con: &RawConnection,
49        sql: &str,
50    ) -> Result<Self> {
51        let mut stmt: duckdb_prepared_statement = ptr::null_mut();
52        // Convert SQL string to C-compatible string for FFI.
53        let c_str = std::ffi::CString::new(sql).expect("Failed to create CString from SQL");
54        // SAFETY: con.con and c_str are valid, stmt is a valid pointer for output.
55        let resp = unsafe { duckdb_prepare(con.con, c_str.as_ptr(), &mut stmt) };
56        // The result_from_duckdb_prepare will check the response and return an error if needed.
57        result_from_duckdb_prepare(resp, stmt).map(|_| Self {
58            con: Arc::new(con.clone()),
59            stmt,
60            result: None,
61            bind_idx: 0,
62        })
63    }

105// Exposed API
106impl Statement {
107    /// Bind a value to the next parameter index using the provided binder.
108    ///
109    /// # Arguments
110    ///
111    /// * `binder` - The value to bind, must implement `AppendAble`.
112    ///
113    /// # Returns
114    ///
115    /// * `Result<()>` - Ok if successful, error otherwise.
116    ///
117    /// # Example
118    ///
119    /// ```ignore
120    /// let mut stmt = Statement::new(&con, "SELECT ?").unwrap();
121    /// let mut binder = MyAppendableValue::new(42);
122    /// stmt.bind(&mut binder).unwrap();
123    /// ```
124    #[allow(unused)]
125    #[inline]
126    pub fn bind<T: AppendAble>(
127        &mut self,
128        binder: &mut T,
129    ) -> Result<()> {
130        self.bind_idx += 1;
131        // Bind at the current index (0-based).
132        self.bind_at(binder, self.bind_idx - 1)
133    }
134
135    /// Bind a value to a specific parameter index using the provided binder.
136    ///
137    /// # Arguments
138    ///
139    /// * `binder` - The value to bind, must implement `AppendAble`.
140    /// * `idx` - The parameter index to bind at (0-based).
141    ///
142    /// # Returns
143    ///
144    /// * `Result<()>` - Ok if successful, error otherwise.
145    ///
146    /// # Example
147    ///
148    /// ```ignore
149    /// let mut stmt = Statement::new(&con, "SELECT ?").unwrap();
150    /// let mut binder = MyAppendableValue::new(42);
151    /// stmt.bind_at(&mut binder, 0).unwrap();
152    /// ```
153    #[allow(unused)]
154    #[inline]
155    pub fn bind_at<T: AppendAble>(
156        &self,
157        binder: &mut T,
158        idx: u64,
159    ) -> Result<()> {
160        // Call the binder's stmt_append to bind the value at the given index.
161        binder.stmt_append(idx, self.stmt)
162    }
163
164    /// Execute the prepared statement and fetch the result.
165    ///
166    /// # Returns
167    ///
168    /// * `Result<Option<RawResult>>` - The result if successful, or an error.
169    ///
170    /// # Errors
171    ///
172    /// Returns an error if the statement has already been executed.
173    ///
174    /// # Example
175    ///
176    /// ```ignore
177    /// let mut stmt = Statement::new(&con, "SELECT 1").unwrap();
178    /// let result = stmt.fetch();
179    /// assert!(result.is_ok());
180    /// ```
181    #[allow(unused)]
182    pub fn fetch(&mut self) -> Result<Option<RawResult>> {
183        if self.result.is_some() {
184            // Prevent double execution of the same statement instance.
185            return Err(crate::error::Error::DuckDBFailure(
186                crate::ffi::Error::new(crate::ffi::DuckDBError),
187                Some("Statement already executed".to_owned()),
188            ));
189        }
190        // SAFETY: result must be zeroed before passing to FFI.
191        let mut result: duckdb_result = unsafe { mem::zeroed() };
192        // SAFETY: stmt must be a valid prepared statement pointer.
193        let resp = unsafe { duckdb_execute_prepared(self.stmt, &mut result) };
194        // The result_from_duckdb_result will check the response and return an error if needed.
195        result_from_duckdb_result(resp, &mut result).map(|_| Some(RawResult::new(result)))
196    }
197
198    /// Get the number of parameters bound to the prepared statement.
199    ///
200    /// # Returns
201    ///
202    /// * `usize` - The number of parameters bound.
203    ///
204    /// # Example
205    ///
206    /// ```ignore
207    /// let mut stmt = Statement::new(&con, "SELECT ?").unwrap();
208    /// let mut binder = MyAppendableValue::new(42);
209    /// stmt.bind(&mut binder).unwrap();
210    /// let count = stmt.bind_parameter_count();
211    /// assert_eq!(count, 1);
212    /// ```
213    #[allow(unused)]
214    #[inline]
215    pub fn bind_parameter_count(&self) -> usize {
216        unsafe { duckdb_nparams(self.stmt) as usize }
217    }
218
219    /// Clear all bindings for the prepared statement.
220    ///
221    /// # Returns
222    ///
223    /// * `Result<()>` - Ok if successful, error otherwise.
224    ///
225    /// # Example
226    ///
227    /// ```ignore
228    /// let mut stmt = Statement::new(&con, "SELECT ?").unwrap();
229    /// let mut binder = MyAppendableValue::new(42);
230    /// stmt.bind(&mut binder).unwrap();
231    /// stmt.clear_bindings().unwrap();
232    /// ```
233    #[allow(unused)]
234    #[inline]
235    pub fn clear_bindings(&self) -> Result<()> {
236        let res = unsafe { duckdb_clear_bindings(self.stmt) };
237        if res != DuckDBSuccess {
238            Err(Error::DuckDBFailure(
239                crate::ffi::Error::new(crate::ffi::DuckDBError),
240                Some("Failed to clear bindings".to_owned()),
241            ))
242        } else {
243            Ok(())
244        }
245    }
246
247    /// Check if the prepared statement is null (not initialized).
248    ///
249    /// # Returns
250    ///
251    /// * `bool` - True if the statement is null, false otherwise.
252    #[allow(unused)]
253    #[inline]
254    pub fn is_null(&self) -> bool {
255        self.stmt.is_null()
256    }
257
258    /// Reset the result of the last execution, if any.
259    ///
260    /// # Safety
261    ///
262    /// Resources held by the previous result (such as DuckDB result pointers)
263    /// are freed automatically when the RawResult is dropped.
264    /// This ensures no memory leaks occur when resetting the result.
265    #[allow(unused)]
266    #[inline]
267    pub fn reset_result(&mut self) {
268        if self.result.is_some() {
269            self.result = None;
270        }
271    }
272}
273/// Clone the statement, but do not clone the result.
274///
275/// NOTE: The underlying prepared statement pointer is not duplicated at the FFI level,
276/// so use with caution. This is a shallow clone.
277impl Clone for Statement {
278    fn clone(&self) -> Self {
279        // TODO: Should we clone the result? or empty the result?
280        Self { con: Arc::clone(&self.con), stmt: self.stmt, result: None, bind_idx: self.bind_idx }
281    }
282}
283
284/// Ensure the prepared statement is destroyed when the Statement is dropped.
285///
286/// # Safety
287///
288/// This will call the DuckDB FFI to destroy the prepared statement. Do not use the pointer after drop.
289impl Drop for Statement {
290    fn drop(&mut self) {
291        unsafe {
292            if !self.stmt.is_null() {
293                // SAFETY: This will free the prepared statement at the FFI level.
294                // Do not use self.stmt after this call.
295                duckdb_destroy_prepare(&mut self.stmt);
296            }
297        }
298    }
