8use super::result::DuckResult;
9use crate::{error::Result, ffi};
10
11pub struct DataChunk(
12    pub(crate) duckdb_data_chunk,
13    pub(crate) u64, // current row index in chunk
14);
15
16impl DataChunk {
17    #[inline]
18    pub fn new(data_chunk: ffi::duckdb_data_chunk) -> Result<DataChunk> {
19        if data_chunk.is_null() {
20            return Err(crate::error::Error::DuckDBFailure(
21                ffi::Error::new(ffi::DuckDBError),
22                Some("data chunk is null".to_owned()),
23            ));
24        }
25        Ok(DataChunk(data_chunk, 0))
26    }
27    #[inline]
28    pub fn from_result(result: &DuckResult) -> Option<Result<DataChunk>> {
29        // SAFETY: `result` is a valid duckdb_result; the returned chunk (if non-null)
30        // is owned by us and must be destroyed via `duckdb_destroy_data_chunk`.
31        let data_chunk = unsafe { ffi::duckdb_fetch_chunk(**result) };
32        if data_chunk.is_null() {
33            return None;
34        }
35        // SAFETY: `data_chunk` is non-null and freshly obtained from `duckdb_fetch_chunk`.
36        let res = DataChunk::new(data_chunk);
37        Some(res)
38    }
39
40    #[allow(unused)]
41    #[inline]
42    pub fn current_row(&self) -> u64 {
43        self.1
44    }
45    #[inline]
46    pub fn row_count(&self) -> u64 {
47        // SAFETY: `self.0` is a valid duckdb_data_chunk (enforced by the caller).
48        unsafe { ffi::duckdb_data_chunk_get_size(self.0) }
49    }
50
51    #[inline]
52    pub fn next_row(&mut self) -> Option<u64> {
53        // SAFETY: `self.0` is a valid duckdb_data_chunk (enforced by the caller).
54        if self.row_count() < 1 {
55            return None;
56        }
57        // SAFETY: same as above.
58        if self.1 >= self.row_count() {
59            // Reset the row index and fetch the next chunk
60            self.1 = 0;
61            // SAFETY: `self.0` is a valid, non-null duckdb_data_chunk; after destroy
62            // we null it so this path is never re-entered.
63            unsafe { duckdb_destroy_data_chunk(&mut (self.0)) };
64            self.0 = ptr::null_mut();
65            return None;
66        }
67        self.1 += 1;
68        Some(self.1 - 1)
69    }
70}
71
72impl Deref for DataChunk {
73    type Target = duckdb_data_chunk;
74
75    #[inline]
76    fn deref(&self) -> &Self::Target {
77        &self.0
78    }
79}
80
81impl DerefMut for DataChunk {
82    #[inline]
83    fn deref_mut(&mut self) -> &mut Self::Target {
84        &mut self.0
85    }
86}
87
88impl Drop for DataChunk {
89    fn drop(&mut self) {
90        if !self.0.is_null() {
91            // SAFETY: `self.0` is a valid non-null `duckdb_data_chunk`. The null guard
92            // ensures this path runs at most once.
93            unsafe { duckdb_destroy_data_chunk(&mut (self.0)) };
94        }
95    }
96}
