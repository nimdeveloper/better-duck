1use crate::types::value::DuckValue;
2
3/// Implementation of methods for the `AbstractRow` struct.
4///
5/// Provides functionality to create a new row from a vector of `DuckValue`s and a list of column names,
6/// as well as to retrieve a value by its column name.
7///
8/// # Methods
9///
10/// - `new(result: Vec<DuckValue>, col_names: Box<[&'static str]>) -> Self`  
11///   Constructs a new `AbstractRow` from the given values and column names.
12///
13/// - `get(&self, name: &str) -> Option<&DuckValue>`  
14///   Retrieves a reference to the value associated with the specified column name, if it exists.
15#[derive(Debug)]
16pub struct AbstractRow(Vec<DuckValue>, Box<[&'static str]>);
17
18impl AbstractRow {
19    /// Creates a new `AbstractRow` instance from a vector of `DuckValue` and a boxed slice of column names.
20    ///
21    /// # Arguments
22    ///
23    /// * `result` - A vector containing the values for each column in the row.
24    /// * `col_names` - A boxed slice of static string slices representing the column names.
25    ///
26    /// # Returns
27    ///
28    /// A new `AbstractRow` containing the provided values and column names.
29    pub fn new(
30        result: Vec<DuckValue>,
31        col_names: Box<[&'static str]>,
32    ) -> Self {
33        AbstractRow(result, col_names)
34    }
35
36    /// Retrieves a reference to the value associated with the specified column name.
37    ///
38    /// # Arguments
39    ///
40    /// * `name` - The name of the column to retrieve the value for.
41    ///
42    /// # Returns
43    ///
44    /// An `Option` containing a reference to the `DuckValue` if the column exists, or `None` if it does not.
45    #[allow(unused)]
46    pub fn get(
47        &self,
48        name: &str,
49    ) -> Option<&DuckValue> {
50        for (i, each) in self.1.iter().enumerate() {
51            if each == &name {
52                return Some(&self.0[i]);
53            }
54        }
55        None
56    }
57}
