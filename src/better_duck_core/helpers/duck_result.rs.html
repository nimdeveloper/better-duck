14/// Converts a DuckDB error code and optional message into a `Result<()>` with a `DuckDBFailure` error.
15///
16/// # Arguments
17///
18/// * `code` - The DuckDB error code returned by an FFI call.
19/// * `message` - An optional error message to include.
20///
21/// # Returns
22///
23/// * `Err(Error::DuckDBFailure)` containing the error code and message.
24///
25/// # Example
26///
27/// ```rust
28/// let result = error_from_duckdb_code(ffi::DuckDBError, Some("Some error".to_string()));
29/// assert!(result.is_err());
30/// ```
31#[inline]
32fn error_from_duckdb_code(
33    code: duckdb_state,
34    message: Option<String>,
35) -> Result<()> {
36    Err(Error::DuckDBFailure(FFIError::new(code), message))
37}

39/// Converts the result of a DuckDB appender operation into a `Result<()>`.
40///
41/// If the operation was successful, returns `Ok(())`. Otherwise, retrieves the error message
42/// from the appender, destroys the appender, and returns an error.
43///
44/// # Arguments
45///
46/// * `code` - The DuckDB state code returned by the appender operation.
47/// * `appender` - A pointer to the DuckDB appender.
48///
49/// # Returns
50///
51/// * `Ok(())` if the operation was successful.
52/// * `Err(Error::DuckDBFailure)` with the error message if the operation failed.
53///
54/// # Example
55///
56/// ```rust
57/// let code = ffi::DuckDBSuccess;
58/// let appender: *mut ffi::duckdb_appender = std::ptr::null_mut();
59/// let result = result_from_duckdb_appender(code, appender);
60/// assert!(result.is_ok());
61/// ```
62#[cold]
63#[inline]
64pub fn result_from_duckdb_appender(
65    code: duckdb_state,
66    appender: *mut duckdb_appender,
67) -> Result<()> {
68    if code == DuckDBSuccess {
69        return Ok(());
70    }
71    unsafe {
72        let message = if (*appender).is_null() {
73            Some("appender is null".to_string())
74        } else {
75            let c_err = duckdb_appender_error(*appender);
76            let message = Some(CStr::from_ptr(c_err).to_string_lossy().to_string());
77            duckdb_appender_destroy(appender);
78            message
79        };
80        error_from_duckdb_code(code, message)
81    }
82}
