1//! UNION read/write helpers.
2//!
3//! DuckDB `UNION` is physically a `STRUCT` where child 0 is the tag discriminant
4//! and children `1..=N` are the member values.  Only the active member's value is
5//! read; the rest are skipped.
6// FFI pointer arguments are used safely inside `unsafe` blocks.
7#![allow(clippy::not_unsafe_ptr_arg_deref)]
8

14        duckdb_union_type_member_count, duckdb_value, duckdb_vector, duckdb_vector_get_column_type,
15        duckdb_vector_get_data, idx_t,
16    },
17};

21// Read path
22
23/// Read a DuckDB `UNION` vector column at `row_idx`.
24///
25/// Returns `DuckValue::Union(Box::new(active_member_value))`.
26///
27/// # Safety
28/// `val` must be a valid `duckdb_vector` of UNION type; `row_idx` must be within
29/// `[0, chunk_size)`.
30pub(crate) fn read_union(
31    val: duckdb_vector,
32    row_idx: u64,
33) -> Result<DuckValue, DuckDBConversionError> {
34    // SAFETY: `val` is a valid union vector.  The logical type is heap-allocated
35    // and must be destroyed exactly once.
36    let mut lt = unsafe { duckdb_vector_get_column_type(val) };
37    // SAFETY: `lt` is a valid logical type of UNION kind.
38    let member_count: idx_t = unsafe { duckdb_union_type_member_count(lt) };
39
40    // Child 0 of the underlying struct layout is the tag vector.
41    // SAFETY: `val` is a valid union vector; child 0 is the tag.
42    let tag_vec = unsafe { duckdb_struct_vector_get_child(val, 0) };
43
44    // DuckDB uses UTINYINT (u8) for ≤ 255 members, USMALLINT (u16) otherwise.
45    // SAFETY: `tag_vec` is a valid data vector; `row_idx` is within [0, chunk_size).
46    let tag: idx_t = unsafe {
47        let data = duckdb_vector_get_data(tag_vec);
48        if member_count <= u8::MAX as idx_t {
49            *(data as *const u8).add(row_idx as usize) as idx_t
50        } else {
51            *(data as *const u16).add(row_idx as usize) as idx_t
52        }
53    };
54
55    if tag >= member_count {
56        // SAFETY: `lt` must be destroyed even on this error path.
57        unsafe { duckdb_destroy_logical_type(&mut lt) };
58        return Err(DuckDBConversionError::ConversionError(format!(
59            "union tag {tag} out of range (member count {member_count})"
60        )));
61    }
62
63    // The active member sits at child index (tag + 1) in the underlying struct.
64    // SAFETY: `val` is a valid union vector; `tag + 1` is within [1, member_count + 1).
65    let member_vec = unsafe { duckdb_struct_vector_get_child(val, tag + 1) };
66    // SAFETY: `member_vec` is a valid vector for the active member.
67    let mut member_lt = unsafe { duckdb_vector_get_column_type(member_vec) };
68    // SAFETY: `member_lt` is a valid logical type.
69    let member_tid = unsafe { crate::ffi::duckdb_get_type_id(member_lt) };
70    // SAFETY: `member_lt` was returned by `duckdb_vector_get_column_type`.
71    unsafe { duckdb_destroy_logical_type(&mut member_lt) };
72
73    // Recurse into the active member.
74    let inner = DuckValue::from_duckdb_vec(member_vec, member_tid, row_idx);
75
76    // SAFETY: `lt` was returned by `duckdb_vector_get_column_type` and must be destroyed once.
77    unsafe { duckdb_destroy_logical_type(&mut lt) };
78
79    inner.map(|v| DuckValue::Union(Box::new(v)))
80}

82// Write path

84/// Build a `duckdb_value` of type `UNION` wrapping the single `inner` member.
85///
86/// Creates a single-member UNION type with member name `"value"` and tag index 0.
87/// The caller is responsible for destroying the returned value.
88pub(crate) fn union_to_duck(inner: &DuckValue) -> Result<duckdb_value, DuckDBConversionError> {
89    let mut member_lt = DuckValue::logical_type_of(inner)?;
90    let c_name = std::ffi::CString::new("value").unwrap();
91    let mut name_ptr: *const std::os::raw::c_char = c_name.as_ptr();
92    // SAFETY: single-element arrays of valid pointers; create copies both.
93    let mut union_lt = unsafe { duckdb_create_union_type(&mut member_lt, &mut name_ptr, 1) };
94    // SAFETY: `member_lt` was allocated by `logical_type_of` above.
95    unsafe { duckdb_destroy_logical_type(&mut member_lt) };
96    let mut member_dv = match inner.to_duck() {
97        Ok(v) => v,
98        Err(e) => {
99            // SAFETY: `union_lt` was allocated above; destroy once.
100            unsafe { duckdb_destroy_logical_type(&mut union_lt) };
101            return Err(e);
102        },
103    };
104    // SAFETY: `union_lt` valid; tag_index=0 (single-member union); `member_dv` valid.
105    let result = unsafe { duckdb_create_union_value(union_lt, 0, member_dv) };
106    // SAFETY: `union_lt` was allocated above; destroy once.
107    unsafe { duckdb_destroy_logical_type(&mut union_lt) };
108    // SAFETY: `member_dv` was created by `to_duck()` above; destroy once.
109    unsafe { duckdb_destroy_value(&mut member_dv) };
110    Ok(result)
111}

115/// Return a `duckdb_logical_type` for a single-member UNION wrapping `inner`.
116pub(crate) fn union_logical_type(
117    inner: &DuckValue
118) -> Result<duckdb_logical_type, DuckDBConversionError> {
119    let mut member_lt = DuckValue::logical_type_of(inner)?;
120    let c_name = std::ffi::CString::new("value").unwrap();
121    let mut name_ptr: *const std::os::raw::c_char = c_name.as_ptr();
122    // SAFETY: single-element arrays of valid pointers; create copies both.
123    let lt = unsafe { duckdb_create_union_type(&mut member_lt, &mut name_ptr, 1) };
124    // SAFETY: `member_lt` was allocated by `logical_type_of` above.
125    unsafe { duckdb_destroy_logical_type(&mut member_lt) };
126    Ok(lt)
127}
