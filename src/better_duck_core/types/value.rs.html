105// Macro to implement DuckDialect for types
106macro_rules! simple_type_conversion {
107    // This macro expects `val` (duckdb_vector pointer) and `row_idx` (usize)
108    // to be in scope where it is called.
109    // It also expects `Ok` to be a valid return type for the surrounding function.
110    ($row_index:expr, $vector_ptr:expr, $rust_type:expr, $duck_primitive_type:ty) => {{
111        // Get the raw data pointer from the DuckDB vector
112        let data_ptr = unsafe { duckdb_vector_get_data($vector_ptr) };
113        // Cast the raw pointer to a pointer of the expected DuckDB primitive type
114        let values: *mut $duck_primitive_type = data_ptr as *mut $duck_primitive_type;
115        // Dereference the value at the specific row index
116        let primitive_value = unsafe { *values.add($row_index as usize) as $duck_primitive_type };
117        // Wrap the primitive value in your Rust type and return Ok
118        Ok($rust_type(primitive_value))
119    }};
120}

128        val: duckdb_vector,
129        t: duckdb_type,
130        row_idx: u64,
131    ) -> Result<DuckValue, DuckDBConversionError> {
132        // Get data as i64 slice
133        let data_ptr = unsafe { duckdb_vector_get_data(val) as *mut i64 };
134        let validity_ptr = unsafe { duckdb_vector_get_validity(val) };
135        // Check validity
136        let is_valid = unsafe { duckdb_validity_row_is_valid(validity_ptr, row_idx) };
137
138        // Access data if valid
139        if !is_valid {
140            return Ok(DuckValue::Null);
141        }
142
143        let value = unsafe { *data_ptr.add(row_idx as usize) as duckdb_value };
144
145        match t {
146            DUCKDB_TYPE_DUCKDB_TYPE_INVALID => {
147                Err(DuckDBConversionError::ConversionError(String::from("Invalid type!")))
148            },
149            DUCKDB_TYPE_DUCKDB_TYPE_BOOLEAN => {
150                // Ok(DuckValue::Boolean(bool::from_duck(value)?))
151                simple_type_conversion!(row_idx, val, DuckValue::Boolean, bool)
152            },
153            DUCKDB_TYPE_DUCKDB_TYPE_TINYINT => {
154                // Ok(DuckValue::TinyInt(i8::from_duck(value)?))
155                simple_type_conversion!(row_idx, val, DuckValue::TinyInt, i8)
156            },
157            DUCKDB_TYPE_DUCKDB_TYPE_SMALLINT => {
158                // Ok(DuckValue::SmallInt(i16::from_duck(value)?))
159                simple_type_conversion!(row_idx, val, DuckValue::SmallInt, i16)
160            },
161            DUCKDB_TYPE_DUCKDB_TYPE_INTEGER => {
162                // Ok(DuckValue::Int(i32::from_duck(value)?))
163                simple_type_conversion!(row_idx, val, DuckValue::Int, i32)
164            },
165            DUCKDB_TYPE_DUCKDB_TYPE_BIGINT => {
166                simple_type_conversion!(row_idx, val, DuckValue::BigInt, i64)
167                // Ok(DuckValue::BigInt(i64::from_duck(value)?))
168            },
169            DUCKDB_TYPE_DUCKDB_TYPE_HUGEINT => {
170                simple_type_conversion!(row_idx, val, DuckValue::HugeInt, i128)
171                // Ok(DuckValue::HugeInt(i128::from_duck(value)?))
172            },
173            DUCKDB_TYPE_DUCKDB_TYPE_UTINYINT => {
174                simple_type_conversion!(row_idx, val, DuckValue::UTinyInt, u8)
175                // Ok(DuckValue::UTinyInt(u8::from_duck(value)?))
176            },
177            DUCKDB_TYPE_DUCKDB_TYPE_USMALLINT => {
178                simple_type_conversion!(row_idx, val, DuckValue::USmallInt, u16)
179                // Ok(DuckValue::USmallInt(u16::from_duck(value)?))
180            },
181            DUCKDB_TYPE_DUCKDB_TYPE_UINTEGER => {

190                simple_type_conversion!(row_idx, val, DuckValue::UHugeInt, u128)

196            },

293                    let mut tmp = Box::<[DuckValue]>::new_uninit_slice(list_length as usize);
294                    iter_ptr = tmp.as_mut_ptr() as *mut DuckValue;
295                    slice_data = Some(tmp);

326                } else {
327                    Err(DuckDBConversionError::ConversionError(String::from(
328                        "Invalid type for List/Array!",
329                    )))
330                }
331            },

389impl From<DuckValue> for String {
390    fn from(val: DuckValue) -> Self {
391        match val {
392            DuckValue::Text(ref s) => s.clone(),
393            DuckValue::Null => String::new(),
394            _ => panic!("Cannot convert {:?} to String", val),
395        }
396    }

398impl From<DuckValue> for i64 {

400        match val {

407        }
