22/// Trait for converting between DuckDB values and Rust types.
23///
24/// This trait provides methods to safely convert DuckDB values to Rust types and vice versa,
25/// handling type safety and conversion errors. Implementors must ensure that conversions
26/// are performed only between compatible types to avoid undefined behavior.
27///
28/// # Safety
29///
30/// The `from_duck` method assumes that the provided `duckdb_value` matches the expected DuckDB type
31/// for the implementing Rust type. Passing a value of an incorrect type may result in panics or
32/// memory safety issues. Always perform type checking at a higher level before calling this method.
33///
34/// # Errors
35///
36/// Both `from_duck` and `to_duck` return a [`DuckDBConversionError`] if the conversion fails.
37///
38/// # Example
39///
40/// ```rust
41/// use better_duck_core::types::{DuckDialect, DuckDBConversionError};
42///
43/// // Assume `duckdb_value` is a valid DuckDB boolean value obtained from DuckDB.
44/// let duckdb_value: duckdb_value = /* obtain from DuckDB */;
45///
46/// // Convert DuckDB value to Rust bool
47/// let rust_bool = bool::from_duck(duckdb_value)?;
48///
49/// // Convert Rust bool back to DuckDB value
50/// let duckdb_value2 = rust_bool.to_duck()?;
51/// # Ok::<(), DuckDBConversionError>(())
52/// ```
53pub trait DuckDialect
54where
55    Self: Sized,
56{
57    /// Converts a DuckDB value to the implementing Rust type.
58    ///
59    /// # Safety
60    ///
61    /// This method assumes that the provided `duckdb_value` is of the correct DuckDB type
62    /// for the implementing Rust type. Calling this method with a value of the wrong type
63    /// is undefined behavior and may lead to panics or memory safety issues.
64    ///
65    /// # Best Practice
66    ///
67    /// Always ensure that the DuckDB type of the value matches the expected Rust type
68    /// before calling this method. Type checking should be performed at a higher level.
69    ///
70    /// # Example
71    ///
72    /// ```
73    /// use better_duck_core::types::{DuckDialect, DuckDBConversionError};
74    /// // Assume `duckdb_value` is a valid DuckDB boolean value.
75    /// let value: duckdb_value = /* obtained from DuckDB */;
76    /// let rust_bool = bool::from_duck(value)?;
77    /// ```
78    ///
79    /// # Errors
80    ///
81    /// Returns a `DuckDBConversionError` if the conversion fails.
82    fn from_duck(value: duckdb_value) -> Result<Self, DuckDBConversionError>
83    where
84        Self: Sized;
85
86    /// Converts the implementing Rust type to a DuckDB value.
87    ///
88    /// This method transforms the Rust value into a DuckDB-compatible value (`duckdb_value`),
89    /// suitable for insertion or update operations in DuckDB.
90    ///
91    /// # Errors
92    ///
93    /// Returns a [`DuckDBConversionError`] if the conversion fails, for example due to
94    /// unsupported types, precision loss, or null values.
95    ///
96    /// # Example
97    ///
98    /// ```
99    /// use better_duck_core::types::{DuckDialect, DuckDBConversionError};
100    ///
101    /// let rust_bool = true;
102    /// let duckdb_value = rust_bool.to_duck()?;
103    /// # Ok::<(), DuckDBConversionError>(())
104    /// ```
105    fn to_duck(&self) -> Result<duckdb_value, DuckDBConversionError>;
106}

110macro_rules! impl_duck_append_able {
111    ($rust_type:ty, $duck_type:expr, $duck_append_fn:expr, $duck_bind_fn:expr) => {
112        impl AppendAble for $rust_type {
113            fn appender_append(
114                &mut self,
115                appender: crate::ffi::duckdb_appender,
116            ) -> Result<()> {
117                unsafe { $duck_append_fn(appender, *self) };
118                Ok(())
119            }
120            fn stmt_append(
121                &mut self,
122                idx: u64,
123                stmt: crate::ffi::duckdb_prepared_statement,
124            ) -> Result<()> {
125                unsafe { $duck_bind_fn(stmt, idx, *self) };
126                Ok(())
127            }
128        }
129    };
130}
