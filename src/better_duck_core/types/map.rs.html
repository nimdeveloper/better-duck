1//! MAP read/write helpers + [`AppendAble`] impl for `HashMap<DuckValue, DuckValue>`.
2//!
3//! DuckDB `MAP` is physically stored as `LIST<STRUCT(key, value)>`.  The read path
4//! walks the flat STRUCT child vectors using the same validity + recursion pattern as
5//! LIST/ARRAY.  The write path builds a `duckdb_value` via `duckdb_create_map_value`.
6//!
7//! Unlike the previous string-keyed representation, MAP keys are now real `DuckValue`
8//! values, preserving the full DuckDB key type.
9// FFI pointer arguments are used safely inside `unsafe` blocks.
10#![allow(clippy::not_unsafe_ptr_arg_deref)]

29// Read path

31/// Read a DuckDB `MAP` vector column at `row_idx`.
32///
33/// MAP is physically `LIST<STRUCT(key, value)>`.  Keys are returned as real
34/// [`DuckValue`] instances — no stringification.
35///
36/// # Safety
37/// `val` must be a valid `duckdb_vector` of MAP type; `row_idx` must be within
38/// `[0, chunk_size)`.
39pub(crate) fn read_map(
40    val: duckdb_vector,
41    row_idx: u64,
42) -> Result<DuckValue, DuckDBConversionError> {
43    // SAFETY: MAP data layout is identical to LIST: each row slot holds a
44    // `duckdb_list_entry { offset, length }`.
45    let data_ptr = unsafe { duckdb_vector_get_data(val) as *const duckdb_list_entry };
46    // SAFETY: `row_idx` is within [0, chunk_size).
47    let entry: duckdb_list_entry = unsafe { *data_ptr.add(row_idx as usize) };
48
49    // `entries_vec` is the flat STRUCT(key, value) child vector.
50    // SAFETY: `val` is a valid MAP/LIST vector.
51    let entries_vec = unsafe { duckdb_list_vector_get_child(val) };
52    // SAFETY: `entries_vec` is a valid STRUCT vector; child 0 = keys.
53    let key_vec = unsafe { duckdb_struct_vector_get_child(entries_vec, 0) };
54    // SAFETY: `entries_vec` is a valid STRUCT vector; child 1 = values.
55    let val_vec = unsafe { duckdb_struct_vector_get_child(entries_vec, 1) };
56
57    // SAFETY: `key_vec` is a valid vector.
58    let mut key_lt = unsafe { duckdb_vector_get_column_type(key_vec) };
59    // SAFETY: `key_lt` was returned by `duckdb_vector_get_column_type`.
60    let key_tid = unsafe { duckdb_get_type_id(key_lt) };
61    // SAFETY: `key_lt` was allocated by `duckdb_vector_get_column_type` above.
62    unsafe { duckdb_destroy_logical_type(&mut key_lt) };
63
64    // SAFETY: `val_vec` is a valid vector.
65    let mut vlt = unsafe { duckdb_vector_get_column_type(val_vec) };
66    // SAFETY: `vlt` was returned by `duckdb_vector_get_column_type`.
67    let val_tid = unsafe { duckdb_get_type_id(vlt) };
68    // SAFETY: `vlt` was allocated by `duckdb_vector_get_column_type` above.
69    unsafe { duckdb_destroy_logical_type(&mut vlt) };
70
71    let mut pairs: HashMap<DuckValue, DuckValue> = HashMap::with_capacity(entry.length as usize);
72    let mut read_err: Option<DuckDBConversionError> = None;
73
74    for j in entry.offset..entry.offset + entry.length {
75        let k = match DuckValue::from_duckdb_vec(key_vec, key_tid, j) {
76            Ok(v) => v,
77            Err(e) => {
78                read_err = Some(e);
79                break;
80            },
81        };
82        let v = match DuckValue::from_duckdb_vec(val_vec, val_tid, j) {
83            Ok(v) => v,
84            Err(e) => {
85                read_err = Some(e);
86                break;
87            },
88        };
89        pairs.insert(k, v);
90    }
91
92    match read_err {
93        Some(e) => Err(e),
94        None => Ok(DuckValue::Map(pairs)),
95    }
96}

98// Write path

100/// Build a `duckdb_value` of type `MAP` from a `HashMap<DuckValue, DuckValue>`.
101///
102/// Returns an error for empty maps (value type cannot be inferred).
103/// The caller is responsible for destroying the returned value.
104pub(crate) fn map_to_duck(
105    m: &HashMap<DuckValue, DuckValue>
106) -> Result<duckdb_value, DuckDBConversionError> {
107    let n = m.len();
108    if n == 0 {
109        return Err(DuckDBConversionError::ConversionError(
110            "cannot convert empty Map to duckdb_value: value type unknown".into(),
111        ));
112    }
113    let pairs: Vec<(&DuckValue, &DuckValue)> = m.iter().collect();
114    let mut key_lt = DuckValue::logical_type_of(pairs[0].0)?; // mut needed for duckdb_destroy_logical_type
115    let mut val_lt = match DuckValue::logical_type_of(pairs[0].1) {
116        Ok(lt) => lt,
117        Err(e) => {
118            // SAFETY: `key_lt` was allocated above.
119            unsafe { duckdb_destroy_logical_type(&mut key_lt) };
120            return Err(e);
121        },
122    };
123    // SAFETY: both types are valid; `duckdb_create_map_type` copies them.
124    let mut map_lt = unsafe { duckdb_create_map_type(key_lt, val_lt) };
125    // SAFETY: `key_lt` was allocated by `logical_type_of` above; destroy once.
126    unsafe { duckdb_destroy_logical_type(&mut key_lt) };
127    // SAFETY: `val_lt` was allocated by `logical_type_of` above; destroy once.
128    unsafe { duckdb_destroy_logical_type(&mut val_lt) };
129
130    let mut key_dvs: Vec<duckdb_value> = Vec::with_capacity(n);
131    let mut val_dvs: Vec<duckdb_value> = Vec::with_capacity(n);
132    let mut err: Option<DuckDBConversionError> = None;
133    for (k, v) in &pairs {
134        match k.to_duck() {
135            Ok(kv) => key_dvs.push(kv),
136            Err(e) => {
137                err = Some(e);
138                break;
139            },
140        }
141        match v.to_duck() {
142            Ok(vv) => val_dvs.push(vv),
143            Err(e) => {
144                err = Some(e);
145                break;
146            },
147        }
148    }
149    if let Some(e) = err {
150        for mut kv in key_dvs {
151            // SAFETY: each `kv` was created by `to_duck()` above; destroy once.
152            unsafe { duckdb_destroy_value(&mut kv) };
153        }
154        for mut vv in val_dvs {
155            // SAFETY: each `vv` was created by `to_duck()` above; destroy once.
156            unsafe { duckdb_destroy_value(&mut vv) };
157        }
158        // SAFETY: `map_lt` was allocated above; destroy once.
159        unsafe { duckdb_destroy_logical_type(&mut map_lt) };
160        return Err(e);
161    }
162    // SAFETY: `map_lt` valid; key/val arrays have `n` elements each.
163    let result = unsafe {
164        duckdb_create_map_value(map_lt, key_dvs.as_mut_ptr(), val_dvs.as_mut_ptr(), n as idx_t)
165    };
166    // SAFETY: `map_lt` was allocated above; destroy once.
167    unsafe { duckdb_destroy_logical_type(&mut map_lt) };
168    for mut kv in key_dvs {
169        // SAFETY: each `kv` was created by `to_duck()` above; destroy once.
170        unsafe { duckdb_destroy_value(&mut kv) };
171    }
172    for mut vv in val_dvs {
173        // SAFETY: each `vv` was created by `to_duck()` above; destroy once.
174        unsafe { duckdb_destroy_value(&mut vv) };
175    }
176    Ok(result)
177}

179// Logical-type path

181/// Return a `duckdb_logical_type` for a MAP with key/value types inferred from the
182/// first entry.
183pub(crate) fn map_logical_type(
184    m: &HashMap<DuckValue, DuckValue>
185) -> Result<duckdb_logical_type, DuckDBConversionError> {
186    if m.is_empty() {
187        return Err(DuckDBConversionError::ConversionError(
188            "cannot determine value type of empty Map".into(),
189        ));
190    }
191    let (first_k, first_v) = m.iter().next().unwrap();
192    let mut key_lt = DuckValue::logical_type_of(first_k)?;
193    let mut val_lt = match DuckValue::logical_type_of(first_v) {
194        Ok(lt) => lt,
195        Err(e) => {
196            // SAFETY: `key_lt` was allocated by `logical_type_of` above; destroy once.
197            unsafe { duckdb_destroy_logical_type(&mut key_lt) };
198            return Err(e);
199        },
200    };
201    // SAFETY: both types are valid; `duckdb_create_map_type` copies them.
202    let lt = unsafe { duckdb_create_map_type(key_lt, val_lt) };
203    // SAFETY: `key_lt` was allocated by `logical_type_of` above; destroy once.
204    unsafe { duckdb_destroy_logical_type(&mut key_lt) };
205    // SAFETY: `val_lt` was allocated by `logical_type_of` above; destroy once.
206    unsafe { duckdb_destroy_logical_type(&mut val_lt) };
207    Ok(lt)
208}

212/// Bind/append a `HashMap<DuckValue, DuckValue>` as a DuckDB `MAP`.
213impl AppendAble for HashMap<DuckValue, DuckValue> {
214    fn stmt_append(
215        &mut self,
216        idx: u64,
217        stmt: crate::ffi::duckdb_prepared_statement,
218    ) -> Result<()> {
219        let mut dv = DuckValue::Map(self.clone()).to_duck().map_err(Error::ConversionError)?;
220        // SAFETY: `stmt`/`idx` are valid; `dv` was created by `to_duck()`.
221        unsafe { duckdb_bind_value(stmt, idx, dv) };
222        // SAFETY: `dv` was created above; destroy exactly once.
223        unsafe { duckdb_destroy_value(&mut dv) };
224        Ok(())
225    }
226
227    fn appender_append(
228        &mut self,
229        appender: crate::ffi::duckdb_appender,
230    ) -> Result<()> {
231        let mut dv = DuckValue::Map(self.clone()).to_duck().map_err(Error::ConversionError)?;
232        // SAFETY: `appender` is valid; `dv` was created by `to_duck()`.
233        unsafe { duckdb_append_value(appender, dv) };
234        // SAFETY: `dv` was created above; destroy exactly once.
235        unsafe { duckdb_destroy_value(&mut dv) };
236        Ok(())
237    }
238}

240/// Order-independent hash for map/struct entries: XOR of per-entry hashes.
241///
242/// We hash each `(k, v)` pair with a fresh `DefaultHasher`, then XOR-fold the
243/// results so that insertion order does not affect the combined hash.
244pub(super) fn map_entries_hash<H: Hasher, K: Hash, V: Hash>(
245    entries: impl Iterator<Item = (K, V)>,
246    len: usize,
247    state: &mut H,
248) {
249    len.hash(state);
250    let xor_fold: u64 = entries
251        .map(|(k, v)| {
252            let mut h = DefaultHasher::new();
253            k.hash(&mut h);
254            v.hash(&mut h);
255            h.finish()
256        })
257        .fold(0u64, |acc, x| acc ^ x);
258    xor_fold.hash(state);
259}

262pub(super) fn map_entries_hash_ref<'a, H: Hasher>(

264    len: usize,
