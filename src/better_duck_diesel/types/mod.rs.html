30/// Implementation of `FromSql` for bool values.
31impl FromSql<Bool, DuckDb> for bool {
32    fn from_sql(val: DuckValueRef<'_>) -> deserialize::Result<Self> {
33        match val {
34            DuckValueRef::Boolean(v) => Ok(v),
35            _ => Err("Unexpected data for boolean type".into()),
36        }
37    }
38}
39
40/// Implementation of `ToSql` for bool values.
41impl ToSql<Bool, DuckDb> for bool {
42    fn to_sql<'b>(
43        &'b self,
44        out: &mut Output<'b, '_, DuckDb>,
45    ) -> serialize::Result {
46        out.set_value(DuckValueRef::Boolean(*self));
47        Ok(IsNull::No)
48    }
49}
