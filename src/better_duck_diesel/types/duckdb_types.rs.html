1//! DuckDB-specific SQL types not available in Diesel's standard type set.
2//!
3//! Diesel's built-in SQL types cover the common SQL standard. DuckDB adds several
4//! unsigned integer widths, `HUGEINT`/`UHUGEINT`, and `TIMESTAMPTZ`. These types
5//! are defined here so that `HasSqlType` and `FromSql`/`ToSql` impls can reference
6//! them without colliding with Diesel's own types.
7
8use diesel_derives::{QueryId, SqlType};
9
10macro_rules! duck_sql_type {
11    ($name:ident, $doc:literal) => {
12        #[doc = $doc]
13        #[derive(Debug, Clone, Copy, Default, PartialEq, Eq, QueryId, SqlType)]
14        pub struct $name;
15    };
16}
17
18duck_sql_type!(DuckTinyInt, "DuckDB `TINYINT` (`INT1`) — maps to `i8`.");
19duck_sql_type!(DuckUTinyInt, "DuckDB `UTINYINT` — maps to `u8`.");
20duck_sql_type!(DuckUSmallInt, "DuckDB `USMALLINT` — maps to `u16`.");
21duck_sql_type!(DuckUInt, "DuckDB `UINTEGER` — maps to `u32`.");
22duck_sql_type!(DuckUBigInt, "DuckDB `UBIGINT` — maps to `u64`.");
23duck_sql_type!(DuckHugeInt, "DuckDB `HUGEINT` — maps to `i128`.");
24duck_sql_type!(DuckUHugeInt, "DuckDB `UHUGEINT` — maps to `u128`.");
25duck_sql_type!(
26    DuckTimestamptz,
27    "DuckDB `TIMESTAMPTZ` — a timestamp with time-zone annotation stored in UTC."
28);
29duck_sql_type!(
30    DuckInterval,
31    "DuckDB `INTERVAL` — maps to `chrono::Duration` (chrono feature) or `std::time::Duration`."
32);
33duck_sql_type!(DuckList, "DuckDB `LIST` — a variable-length array of a uniform element type.");
34duck_sql_type!(
35    DuckTimeTz,
36    "DuckDB `TIME_TZ` — a time-of-day value with UTC offset (microsecond precision)."
37);
38duck_sql_type!(DuckTimeNs, "DuckDB `TIME_NS` — a nanosecond-precision time-of-day value.");
39duck_sql_type!(
40    DuckEnum,
41    "DuckDB `ENUM` — a categorical string type backed by a dictionary encoding."
42);
