13/// Deserialize a DuckDB `VARCHAR` column into a [`String`].
14///
15/// DuckDB may represent the result of a `SELECT DISTINCT` (or other aggregation)
16/// over a `VARCHAR` column using dictionary encoding, in which case the physical
17/// type reported by the C API is `ENUM` rather than `VARCHAR`.  Both variants are
18/// accepted here so that the standard Diesel `Text` SQL type works transparently
19/// regardless of whether DuckDB chose to dictionary-encode the result.
20impl FromSql<Text, DuckDb> for String {
21    fn from_sql(val: DuckValueRef) -> deserialize::Result<Self> {
22        match val {
23            DuckValueRef::Text(v) => Ok(v.into_owned()),
24            // DuckDB uses dictionary encoding (ENUM physical type) for DISTINCT /
25            // aggregate results on VARCHAR columns.
26            DuckValueRef::Enum(v) => Ok(v.into_owned()),
27            _ => Err("Unexpected data for string type".into()),
28        }
29    }
30}

32/// Serialize a `str` string slice as a DuckDB `VARCHAR` bind parameter.
33///
34/// Diesel's blanket `impl<A, T: ToSql<A, DB> + ?Sized, DB: Backend> ToSql<A, DB> for &T`
35/// derives `&str: ToSql<Text, DuckDb>` from this impl automatically.
36/// Similarly, Diesel's `impl<DB> ToSql<Text, DB> for String where str: ToSql<Text, DB>`
37/// derives `String: ToSql<Text, DuckDb>`.
38/// Having only the `str` impl avoids conflicting with those blankets.
39impl ToSql<Text, DuckDb> for str {
40    fn to_sql<'b>(
41        &'b self,
42        out: &mut Output<'b, '_, DuckDb>,
43    ) -> serialize::Result {
44        out.set_value(DuckValueRef::Text(Cow::Borrowed(self)));
45        Ok(IsNull::No)
46    }
47}

51/// Deserialize a DuckDB `ENUM` column into a [`String`].
52///
53/// DuckDB stores enum values as dictionary-encoded strings. The decoded
54/// label is exposed via `DuckValueRef::Enum(Cow<str>)`.
55impl FromSql<DuckEnum, DuckDb> for String {
56    fn from_sql(val: DuckValueRef) -> deserialize::Result<Self> {
57        match val {
58            DuckValueRef::Enum(v) => Ok(v.into_owned()),
59            _ => Err("Unexpected data for String (ENUM) type".into()),
60        }
61    }
62}
