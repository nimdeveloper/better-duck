1//! `FromSql`/`ToSql` implementations for DuckDB `LIST` / Diesel `DuckList`.
2//!
3//! The Rust mirror type is `Vec<DuckValue>`, which lets callers work with any
4//! element type that DuckDB supports without needing separate monomorphisations
5//! for every element variety. Element conversion is handled by
6//! `DuckValue::from` / `DuckValueRef::from`.
7
8use better_duck_core::types::{value::DuckValue, value_ref::DuckValueRef};
9use diesel::{
10    deserialize::{self, FromSql},
11    serialize::{self, IsNull, Output, ToSql},
12};
13
14use crate::backend::DuckDb;
15use crate::types::duckdb_types::DuckList;
16
17// FromSql
18
19/// Deserialize a DuckDB `LIST` column into a [`Vec<DuckValue>`].
20///
21/// Each element in the list is converted to its owned [`DuckValue`]
22/// representation, preserving `NULL` elements as [`DuckValue::Null`].
23impl FromSql<DuckList, DuckDb> for Vec<DuckValue> {
24    fn from_sql(val: DuckValueRef<'_>) -> deserialize::Result<Self> {
25        match val {
26            // `From<&DuckValueRef<'_>> for DuckValue` converts each element.
27            DuckValueRef::List(items) => Ok(items.iter().map(DuckValue::from).collect()),
28            other => Err(format!("expected List, got {other:?}").into()),
29        }
30    }
31}
32
33// ToSql
34
35/// Serialize a [`Vec<DuckValue>`] as a DuckDB `LIST` bind parameter.
36///
37/// Each element is converted to a zero-copy (where possible) `DuckValueRef`
38/// for the lifetime `'b` tied to the `self` borrow.
39impl ToSql<DuckList, DuckDb> for Vec<DuckValue> {
40    fn to_sql<'b>(
41        &'b self,
42        out: &mut Output<'b, '_, DuckDb>,
43    ) -> serialize::Result {
44        // Clone each element into an owned DuckValue, then convert to a fully-owned
45        // DuckValueRef (all Cow branches become Cow::Owned, so the lifetime is 'static
46        // and satisfies the 'b bound on the output).
47        let items: Vec<DuckValueRef<'b>> = self.iter().cloned().map(DuckValueRef::from).collect();
48        out.set_value(DuckValueRef::List(items));
49        Ok(IsNull::No)
50    }
51}
