1//! Diesel `Row` and `Field` implementations for DuckDB.
2//!
3//! `Row` owns its [`DuckRow`] data and shares column names via [`Arc`] with the
4//! cursor that produced it. This avoids any self-referential lifetimes and the
5//! unsound `&'a DuckResult<'a>` pattern.
6
7use std::{marker::PhantomData, ops::Range, sync::Arc};
8
9use better_duck_core::{types::value_ref::DuckValueRef, DuckRow};
10use diesel::row::{Field as DieselField, PartialRow, Row as DieselRow, RowIndex, RowSealed};
11
12use crate::backend::DuckDb;
13
14// Field
15
16/// A single field in a DuckDB result row.
17pub struct Field<'f> {
18    /// Column name borrowed from the `Arc`-shared name slice in the owning [`Row`].
19    name: &'f str,
20    /// The field value as a [`DuckValueRef`].
21    value: DuckValueRef<'f>,
22}
23
24impl<'f> DieselField<'f, DuckDb> for Field<'f> {
25    fn field_name(&self) -> Option<&str> {
26        Some(self.name)
27    }
28
29    /// Returns `None` when the field is `NULL`, `Some(value)` otherwise.
30    ///
31    /// Diesel's `Field` contract requires `value()` to return `None` for NULL
32    /// values so that `from_nullable_sql(None)` is called and `Option<T>`
33    /// correctly resolves to `Ok(None)`.  Returning `Some(DuckValueRef::Null)`
34    /// for NULL would bypass that path and propagate the `Null` variant into
35    /// concrete `FromSql` impls, causing spurious deserialization errors.
36    fn value(&self) -> Option<DuckValueRef<'_>> {
37        match &self.value {
38            DuckValueRef::Null => None,
39            v => Some(v.clone()),
40        }
41    }
42
43    fn is_null(&self) -> bool {
44        matches!(self.value, DuckValueRef::Null)
45    }
46}

48// Row
49
50/// A single row from a DuckDB result set.
51///
52/// The lifetime parameter `'conn` ties the row to the connection that produced
53/// it, ensuring the row cannot outlive the cursor (and by extension the
54/// connection).
55pub struct Row<'conn> {
56    inner: DuckRow,
57    col_names: Arc<[Box<str>]>,
58    _marker: PhantomData<&'conn ()>,
59}
60
61impl<'conn> Row<'conn> {
62    /// Creates a new `Row` from a [`DuckRow`] and a shared column-name slice.
63    pub(crate) fn new(
64        inner: DuckRow,
65        col_names: Arc<[Box<str>]>,
66    ) -> Self {
67        Row { inner, col_names, _marker: PhantomData }
68    }
69}
70
71impl<'conn> DieselRow<'conn, DuckDb> for Row<'conn> {
72    type Field<'f>
73        = Field<'f>
74    where
75        'conn: 'f,
76        Self: 'f;
77
78    type InnerPartialRow = Self;
79
80    fn field_count(&self) -> usize {
81        self.inner.column_count() as usize
82    }
83
84    fn partial_row(
85        &self,
86        range: Range<usize>,
87    ) -> PartialRow<'_, Self> {
88        PartialRow::new(self, range)
89    }
90
91    fn get<'b, I>(
92        &'b self,
93        idx: I,
94    ) -> Option<Self::Field<'b>>
95    where
96        'conn: 'b,
97        Self: RowIndex<I>,
98    {
99        let i = self.idx(idx)?;
100        let name = self.col_names.get(i)?.as_ref();
101        let value = self.inner.get_idx(i).map(DuckValueRef::from)?;
102        Some(Field { name, value })
103    }
104}
105
106impl RowIndex<usize> for Row<'_> {
107    fn idx(
108        &self,
109        i: usize,
110    ) -> Option<usize> {
111        (i < self.inner.column_count() as usize).then_some(i)
112    }
113}
114
115impl<'b> RowIndex<&'b str> for Row<'_> {
116    fn idx(
117        &self,
118        name: &'b str,
119    ) -> Option<usize> {
120        self.col_names.iter().position(|n| n.as_ref() == name)
121    }
122}
