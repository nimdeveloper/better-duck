1//! Error conversion between `better_duck_core` errors and Diesel's error types.
2
3use std::fmt;
4
5/// Wraps a [`better_duck_core::error::Error`] for conversion into a Diesel error.
6pub struct DuckDbError {
7    orig: better_duck_core::error::Error,
8}
9
10impl DuckDbError {
11    /// Creates a new [`DuckDbError`] wrapping a core error.
12    pub fn new(e: better_duck_core::error::Error) -> Self {
13        DuckDbError { orig: e }
14    }
15}
16
17impl fmt::Display for DuckDbError {
18    fn fmt(
19        &self,
20        f: &mut fmt::Formatter<'_>,
21    ) -> fmt::Result {
22        self.orig.fmt(f)
23    }
24}
25
26impl fmt::Debug for DuckDbError {
27    fn fmt(
28        &self,
29        f: &mut fmt::Formatter<'_>,
30    ) -> fmt::Result {
31        write!(f, "DuckDbError({:?})", self.orig)
32    }
33}
34
35impl From<DuckDbError> for diesel::result::Error {
36    fn from(e: DuckDbError) -> Self {
37        use better_duck_core::error::Error as CE;
38        use diesel::result::{DatabaseErrorKind as K, Error as DE};
39
40        match e.orig {
41            CE::QueryReturnedNoRows => DE::NotFound,
42            CE::NulError(n) => DE::InvalidCString(n),
43            CE::Utf8Error(u) => DE::SerializationError(Box::new(u)),
44            CE::ToSqlConversionFailure(b) => DE::SerializationError(b),
45            CE::InvalidColumnIndex(i) => {
46                DE::DeserializationError(format!("invalid column index {i}").into())
47            },
48            CE::InvalidColumnName(n) => {
49                DE::DeserializationError(format!("unknown column '{n}'").into())
50            },
51            CE::DuckDBFailure(_, msg) => {
52                let msg = msg.unwrap_or_else(|| "duckdb error".to_owned());
53                DE::DatabaseError(K::Unknown, Box::new(msg))
54            },
55            other => DE::DatabaseError(K::Unknown, Box::new(format!("{other}"))),
56        }
57    }
58}
